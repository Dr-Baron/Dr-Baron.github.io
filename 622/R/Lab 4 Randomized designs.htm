R example 3 – stratification by gender

 

>  male <- blockrand(n=12, num.levels=3, id.prefix='M', block.prefix='M',stratum='Male')

>  female <- blockrand(n=12, num.levels=3, id.prefix='F', block.prefix='F',stratum='Female')

>  my.study <- rbind(male,female)
